#include <stdio.h>

int main()
{
	int n;
	printf("Quantos numero vocÃª quer analisar: ");
	scanf("%d", &n);
	
	int x, soma_pares = 0;
	
	int quantiaPares 	= 0,
	    quantiaImpares 	= 0;
		
	register int i;
	
	for(i = 1; i <= n; i++)
	{
		printf("valor 0%d: ", i);
		scanf("%d", &x);
		
		if( !(x%2) ) {
		    quantiaPares++;

	}
	
	//Mostra os resultados

	printf("%d Numeros impares:.\n", quantiaImpares);
	printf("%d Foi a soma dos numeros paraes.\n", soma_pares);
	
	return (0);
}
